 15 const crypto = require('crypto');
 16 
 17 class Character {
 18 	constructor(opt) {
 19 		Object.keys(opt).forEach( k => {
 20 			this[`_${k}`] = opt[k]
 21 		})
 22 	}
 23 
 24 	id() {
 25 		const hash = crypto.createHash('sha256');
 26 		hash.update(this.name({}) + this._age);
 27 		return hash.digest('hex');
 28 	}
 29 
 30 	name({name}) {
 31 		let n = this._firstname + ' ' + this._lastname
 32 		if (!name)
 33 			return n
 34 		if (n == name)
 35 			return this
 36 	}
 37 
 38 	texture() {
 39 		return this._texture
 40 	}
 41 
 42 	age() {
 43 		return this._age
 44 	}
 45 
 46 	particularities({particularities}) {
 47 		if (!particularities)
 48 			return this._particularities
 49 		if (particularities && this._particularities.filter(e => e == particularities).length != 0)
 50 			return this
 51 	}
 52 
 53 	relationships({state}) {
 54 		if (!state)
 55 			return this._relationships
 56 		if (state)
 57 			return this._relationships.filter( e => e.state == state )
 58 	}
 59 }
 60 
 61 module.exports = Character
