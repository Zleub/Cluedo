 50 class Character {
 51 	constructor(opt) {
 52 		this._name = opt.name
 53 		this._relationships = opt.relationships
 54 		this._particularities = opt.particularities
 55 	}
 56 
 57 	name({name}) {
 58 		if (!name)
 59 			return this._name
 60 		if (this._name == name)
 61 			return this
 62 	}
 63 
 64 	particularities({particularities}) {
 65 		if (!particularities)
 66 			return this._particularities
 67 		if (particularities && this._particularities.filter(e => e == particularities).length != 0)
 68 			return this
 69 	}
 70 
 71 	relationships({state}) {
 72 		if (!state)
 73 			return this._relationships
 74 		if (state)
 75 			return this._relationships.filter( e => e.state == state )
 76 	}
 77 }
