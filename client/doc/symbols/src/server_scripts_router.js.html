 15 class Router {
 16 	constructor(routes) {
 17 		this.routes = routes
 18 	}
 19 
 20 	matchRoute(req, routes, args = []) {
 21 		return Object.keys(routes).reduce( (p, k) => {
 22 			let match = req.url.match(`^${k}`)
 23 			if (match) {
 24 				let _url = req.url.substr(match[0].length)
 25 				let _f = routes[k][req.method.toLowerCase()]
 26 
 27 				if (_url == "" && _f)
 28 					return function () {
 29 						_f.apply(this, args)
 30 					}
 31 				else {
 32 					return this.matchRoute({
 33 						url: _url,
 34 						method: req.method,
 35 						headers: req.headers
 36 					}, routes[k], args)
 37 				}
 38 			}
 39 			return p
 40 		}, null) || Object.keys(routes).reduce( (p, k) => {
 41 			if (k.indexOf(":") == 1) {
 42 				let _url = req.url.match(/\/(\w+)(\/.+)/)
 43 				let _f = routes[k][req.method.toLowerCase()]
 44 
 45 				if (_f && _url == null)
 46 					return function () {
 47 						_f.apply(this, args.concat([req.url.substr(1)]))
 48 					}
 49 				else {
 50 					args.push(_url[1])
 51 					return this.matchRoute({
 52 						url: _url != null ? _url[2] : req.url,
 53 						method: req.method,
 54 						headers: req.headers
 55 					}, routes[k], args)
 56 				}
 57 			}
 58 
 59 			return p
 60 		}, null)
 61 	}
 62 
 63 	dispatch(req, res, callback) {
 64 		if (req.method === 'HEAD') {
 65 			req.method = 'GET'
 66 		}
 67 
 68 		let route = this.matchRoute(req, this.routes)
 69 
 70 		if (route) {
 71 			route.apply({req, res})
 72 			callback()
 73 		}
 74 		else
 75 			callback(`No such route ${req.method} ${req.url}`)
 76 	}
 77 }
 78 
 79 exports.Router = Router
