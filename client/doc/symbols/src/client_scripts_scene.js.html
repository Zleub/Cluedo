 15 let Scene = (canvas) => {
 16 	let scene = new THREE.Scene()
 17 	let camera = new THREE.PerspectiveCamera( 75, canvas.width / canvas.height, 0.1, 1000 )
 18 
 19 	let renderer = new THREE.WebGLRenderer({
 20 		canvas: canvas
 21 	})
 22 	renderer.shadowMap.enabled = true
 23 	renderer.shadowMap.type = THREE.PCFSoftShadowMap
 24 	renderer.gammaInput = true;
 25 	renderer.gammaOutput = true;
 26 
 27 	var raycaster = new THREE.Raycaster()
 28 
 29 	let texture = new THREE.TextureLoader().load( "assets/block/bedrock.png" )
 30 	texture.magFilter	= THREE.NearestFilter;
 31 	texture.minFilter	= THREE.NearestFilter;
 32 	let geometry = new THREE.BoxGeometry( 1, 1, 1 )
 33 	let material = new THREE.MeshPhongMaterial( {
 34 		color: 0xa5a5a5,
 35 		// map: texture,
 36 		// wireframe: true
 37 	} )
 38 
 39 	for (let i = -4.5; i < 5; i += 1) {
 40 		for (let j = -4.5; j < 5; j += 1) {
 41 			let cube = new THREE.Mesh( geometry, material )
 42 			cube.position.set( i, 0, j)
 43 			cube.castShadow = true
 44 			cube.receiveShadow = true
 45 			scene.add( cube )
 46 		}
 47 	}
 48 
 49 	let vector = new THREE.Vector3()
 50 	let center = new THREE.Vector3()
 51 	let spherical = new THREE.Spherical()
 52 	let normalMatrix = new THREE.Matrix3()
 53 	let rotate = ( delta ) => {
 54 		vector.copy( camera.position ).sub( center );
 55 
 56 		spherical.setFromVector3( vector );
 57 		spherical.theta += delta.x / 200;
 58 		spherical.phi += delta.y / 200;
 59 		spherical.makeSafe();
 60 
 61 		vector.setFromSpherical( spherical );
 62 
 63 		camera.position.copy( center ).add( vector );
 64 		camera.lookAt( center );
 65 	}
 66 
 67 	let pan = function ( delta ) {
 68 		let distance = camera.position.distanceTo( center );
 69 
 70 		delta.multiplyScalar( distance * 0.001 );
 71 		delta.applyMatrix3( normalMatrix.getNormalMatrix( camera.matrix ) );
 72 
 73 		camera.position.add( delta );
 74 		center.add( delta );
 75 	}
 76 
 77 	let STATE = { NONE: - 1, ROTATE: 0, ZOOM: 1, PAN: 2 }
 78 	let state = STATE.NONE
 79 	let pointer = new THREE.Vector2()
 80 	let pointerOld = new THREE.Vector2()
 81 
 82 	canvas.addEventListener('wheel', ( event ) => {
 83 		event.preventDefault();
 84 		zoom( new THREE.Vector3( 0, 0, event.deltaY ) )
 85 	})
 86 
 87 	zoom = function ( delta ) {
 88 		let distance = camera.position.distanceTo( center );
 89 		delta.multiplyScalar( distance * 0.001 );
 90 
 91 		if ( delta.length() > distance )
 92 			return;
 93 
 94 		delta.applyMatrix3( normalMatrix.getNormalMatrix( camera.matrix ) );
 95 		camera.position.add( delta );
 96 	}
 97 
 98 	canvas.addEventListener("dblclick", (e) => {
 99 		let mouse = new THREE.Vector2(e.offsetX / canvas.width * 2 - 1, -(e.offsetY / canvas.height) * 2 + 1)
100 
101 		raycaster.setFromCamera( mouse, camera )
102 
103 		let _ = raycaster.intersectObjects( scene.children, true )[0]
104 
105 		if (_) {
106 			if (_.object.userData.parent && _.object.userData.parent != scene.mainCharacter) {
107 				_.object.userData.parent.lookAt(new THREE.Vector3(scene.mainCharacter.position.x, 1.1, scene.mainCharacter.position.z))
108 				console.log(_.object.userData.parent)
109 				scene.dialog.setTitle(_.object.userData.parent.name)
110 				scene.dialog.toggle()
111 				return
112 			}
113 			else if ( _.object.userData.parent == scene.mainCharacter) {
114 				return
115 			}
116 
117 			let {object} = _
118 			scene.mainCharacter.lookAt( new THREE.Vector3(object.position.x, 1.1, object.position.z) )
119 			scene.mainCharacter.walk()
120 		}
121 	})
122 	canvas.addEventListener("click", (e) => {
123 		let mouse = new THREE.Vector2(e.offsetX / canvas.width * 2 - 1, -(e.offsetY / canvas.height) * 2 + 1)
124 
125 		raycaster.setFromCamera( mouse, camera )
126 
127 		let _ = raycaster.intersectObjects( scene.children )[0]
128 
129 		if (_) {
130 			let {object} = _
131 			scene.mainCharacter.lookAt( new THREE.Vector3(object.position.x, 1.1, object.position.z) )
132 		}
133 	})
134 
135 	canvas.addEventListener("mousedown", ( e ) => {
136 		if ( e.button === 0 ) {
137 			state = STATE.ROTATE
138 		} else if ( event.button === 1 ) {
139 			state = STATE.ZOOM;
140 		} else if ( event.button === 2 ) {
141 			state = STATE.PAN;
142 		}
143 
144 		pointerOld.set( e.offsetX, e.offsetY )
145 
146 		canvas.addEventListener( 'mousemove', onMouseMove, false )
147 		window.addEventListener( 'mouseup', onMouseUp, false )
148 	})
149 
150 	let onMouseUp = () => {
151 		canvas.removeEventListener( 'mousemove', onMouseMove, false )
152 		window.removeEventListener( 'mouseup', onMouseUp, false )
153 	}
154 
155 	let onMouseMove = (e) => {
156 		pointer.set( e.offsetX, e.offsetY )
157 
158 		let movementX = pointer.x - pointerOld.x
159 		let movementY = pointer.y - pointerOld.y
160 
161 		if ( state === STATE.ROTATE ) {
162 			rotate( new THREE.Vector3( - movementX * 1, - movementY * 1, 0 ) )
163 		} else if ( state === STATE.PAN ) {
164 			pan( new THREE.Vector3( - movementX, movementY, 0 ) );
165 		} else if ( state === STATE.ZOOM ) {
166 			zoom( new THREE.Vector3( 0, 0, movementY ) );
167 		}
168 
169 		pointerOld.set( e.offsetX, e.offsetY )
170 
171 	}
172 
173 	canvas.addEventListener( 'contextmenu', ( event ) => {
174 		event.preventDefault();
175 	}, false )
176 
177 	let update_stack = []
178 	let clock = new THREE.Clock()
179 	clock.start()
180 
181 	function render() {
182 		update_stack.forEach( e => e({clock, camera}) )
183 		renderer.render( scene, camera )
184 		requestAnimationFrame( render )
185 	}
186 
187 	camera.position.y = 5
188 	camera.position.z = 30
189 	zoom( new THREE.Vector3( 0, 0, -800 ) )
190 	camera.lookAt( new THREE.Vector3() )
191 
192 	render()
193 
194 	return {scene, camera, renderer, update_stack}
195 }
