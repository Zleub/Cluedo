 15 http = require('http')
 16 https = require('https')
 17 querystring = require('querystring')
 18 fs = require('fs')
 19 
 20 let { Router } = require('./scripts/router.js')
 21 
 22 opt = {
 23 	port: 8080,
 24 	hostname: "0.0.0.0"
 25 }
 26 
 27 process.argv.reduce( (p, e) => {
 28 	if (p == "--token")
 29 		opt.token = e
 30 	if (p == "--port")
 31 		opt.port = Number(e)
 32 	if (p == "--hostname")
 33 		opt.hostname = e
 34 	return e
 35 })
 36 
 37 new Promise( (res, rej) => {
 38 	fs.readdir('./server/routes', (err, data) => {
 39 		res( data.reduce( (p, e) => {
 40 			let n = e.match(/^([^.]\w+)/)
 41 			if (!n)
 42 				return p
 43 			p[`/${n[1]}`] = require(`./routes/${e}`)
 44 			return p
 45 		}, {}) )
 46 	})
 47 } ).then( routes => {
 48 	console.log(routes)
 49 
 50 	let router = new Router(routes)
 51 	let server = http.createServer(function (req, res) {
 52 		req.chunks = [];
 53 
 54 		req.on('data', function (chunk) {
 55 			req.chunks.push(chunk);
 56 		});
 57 
 58 		req.on('end', function () {
 59 			if (req.chunks.length > 0) {
 60 				try {
 61 					req.body = JSON.parse( Buffer.concat(req.chunks).toString() ) || {}
 62 				} catch (e) {
 63 					req.body = {}
 64 				}
 65 			}
 66 
 67 			router.dispatch(req, res, function (err) {
 68 				if (err) {
 69 					console.error(err)
 70 					res.writeHead(404);
 71 					res.end();
 72 				}
 73 				else
 74 					console.log('Served ' + req.url);
 75 			});
 76 		})
 77 
 78 
 79 	});
 80 
 81 	server.listen(opt.port, opt.hostname);
 82 
 83 	console.log('Server Up')
 84 })
