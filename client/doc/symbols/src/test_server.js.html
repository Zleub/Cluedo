 15 const { spawn, spawnSync } = require('child_process');
 16 const assert = require('assert');
 17 
 18 module.exports = () => {
 19 	let server
 20 	new Promise( (res, rej) => {
 21 		server = spawn('node', [
 22 			'server/main.js',
 23 			'--port', '8000',
 24 			'--hostname', 'localhost'
 25 		]);
 26 
 27 		server.stdout.on('data', (data) => {
 28 			process.stdout.write(`\nserver stdout:\n${data}`);
 29 			if (data == "Server Up\n")
 30 				res()
 31 		});
 32 
 33 		server.stderr.on('data', (data) => {
 34 			process.stdout.write(`server stderr:\n${data}`);
 35 		});
 36 
 37 		server.on('close', (code) => {
 38 			process.stdout.write(`server child process exited with code ${code}`);
 39 		});
 40 	}).then( () => {
 41 		const tests = [
 42 			{
 43 				url: '', code: 404,
 44 				response: ''
 45 			}
 46 		]
 47 
 48 		let exit = (code) => {
 49 			server.kill()
 50 			process.exit(code)
 51 		}
 52 
 53 		let testCode = (e) => {
 54 			const {stdout, stderr, status} = spawnSync('curl', ['-I', `localhost:8000${e.url}`])
 55 			let code = stdout.toString().match(/HTTP\/.\..\s*(\d+)/)[1]
 56 
 57 			try {
 58 				assert.equal(code, e.code)
 59 			} catch (_) {
 60 				console.error(`Assert fail: ${e.url} ${_.message.toString()}`)
 61 				exit(1)
 62 			}
 63 		}
 64 
 65 		let testResponse = (e) => {
 66 			const {stdout, stderr, status} = spawnSync('curl', [`localhost:8000${e.url}`])
 67 			try {
 68 				assert.equal(stdout.toString(), e.response)
 69 			} catch (_) {
 70 				console.error(`Assert fail: ${e.url} ${_.message.toString()}`)
 71 				exit(1)
 72 			}
 73 		}
 74 
 75 		tests.forEach( e => {
 76 			testCode(e)
 77 			testResponse(e)
 78 		})
 79 
 80 		exit(0)
 81 	})
 82 }
