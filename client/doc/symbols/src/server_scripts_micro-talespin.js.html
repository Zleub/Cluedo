 17 const _usage = {
 18 	'--verbose': 'this option is verbosity'

 21 let usage = process.argv.reduce( (p, e) => {
 22 	Object.keys(_usage).forEach( _ => {
 23 		if (e == _)
 24 			p[_] = true
 25 	})
 26 	return p
 27 }, {})

 29 function verbose(...s) {
 30 	if (!usage['--verbose'])
 31 		return
 32 
 33 	let _verbose = (s) => {
 34 		if (typeof s == 'string')
 35 			console.log(s)
 36 		else if (typeof s == 'function')
 37 			console.log(s.toString())
 38 		else if (typeof s == 'object')
 39 			console.log(JSON.stringify(s, null, "  "))
 40 		else if (s === true)
 41 			console.log("true")
 42 		else if (s === false)
 43 			console.log("false")
 44 		else if (s === null)
 45 			console.log("false")
 46 		else
 47 			console.log("undefined")
 48 	}
 49 
 50 	s.forEach( _verbose )
 51 }

 53 function facts(actor, action, data) {

 57 function talesFactory(mods) {
 58 	let tale = function ({id, actors, personae, object, initFacts}) {
 59 		this.id = id
 60 		this.personae = personae
 61 		this.object = object
 62 		this.actors = actors
 63 		actors.forEach( e => this[e] = {})
 64 		initFacts.forEach( e => e.call(this) )
 65 	}
 66 	tale.prototype.constructor = tale
 67 	tale.prototype.verbose = function (s) { this._verbose ? verbose(s) : null }
 68 	tale.prototype.run = function (identifier, item) {
 69 		this.verbose('run')
 70 		let count = 0
 71 		while (count < 10) {
 72 			this.verbose( "---- ---- ---- ----".underline )
 73 			this.personae.reduce( (p, e) => {
 74 				let plan =  this[e].stack ? this[e].stack[this[e].stack.length - 1] : undefined
 75 				if (plan) {
 76 					if (typeof plan.action == 'string')
 77 						this.verbose(`${plan.action} not implemented`.red)
 78 					else
 79 						plan.action.functionConsequence(this, e, plan)
 80 				}
 81 
 82 				this.verbose( e.underline )
 83 				this.verbose( this[e].stack )
 84 
 85 				count += 1
 86 				return p + this[e].goals ? this[e].goals.length : 0
 87 			}, 0 )
 88 		}
 89 		console.log( "---- ---- ---- ----" )
 90 		this.printFacts()
 91 	}
 92 	tale.prototype.is = function (identifier, item) {
 93 		return this[identifier] ? this[identifier].some( e => e == item) : false
 94 	}
 95 	tale.prototype.have = function (actor, item) {
 96 		this.verbose('have'.inverse)
 97 		return this[actor].facts.some( e => e.actor == actor && e.data == item )
 98 	}
 99 	tale.prototype.printWorldFacts = function () {
100 		this.facts('World')
101 	}
102 	tale.prototype.printFacts = function (actor) {
103 		let isWorld = (actor) => actor == 'World' ? 'The world' : actor
104 		let print = (actor) => this[actor].facts.forEach( e => {
105 			if (this.is('personae', e.actor)) {
106 				if (actor != e.actor)
107 					console.log(`${isWorld(actor)} thinks that ${e.actor}${e.action.text.personae} ${e.data}`)
108 				else
109 					console.log(`${e.actor}${e.action.text.personae} ${e.data}`)
110 			}
111 			else if (this.is('object', e.actor)) {
112 				console.log(`${isWorld(actor)} thinks that there is ${e.actor} ${e.action.text.object} ${e.data}`)
113 			}
114 		})
115 		if (!actor) {
116 			this.actors.forEach(e => print(e))
117 		}
118 		else
119 			print(actor)
120 	}
121 	tale.prototype.mods = mods
122 	mods.forEach( e => {
123 		tale.prototype[e.name] = e.function.bind(e)
124 		if (e.functionConsequence)
125 			tale.prototype[`e.nameConsequence`] = e.functionConsequence.bind(e)
126 	})
127 	return tale
128 }
