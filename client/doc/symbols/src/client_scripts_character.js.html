161 let applyUV = (map, cube) => {
162 	map.forEach( (e, i) => {
163 		cube.geometry.faceVertexUvs[0][i * 2][0].x = e[0].x * 1 / 64
164 		cube.geometry.faceVertexUvs[0][i * 2][0].y = e[0].y * 1 / 32
165 		cube.geometry.faceVertexUvs[0][i * 2][1].x = e[1].x * 1 / 64
166 		cube.geometry.faceVertexUvs[0][i * 2][1].y = e[1].y * 1 / 32
167 		cube.geometry.faceVertexUvs[0][i * 2][2].x = e[2].x * 1 / 64
168 		cube.geometry.faceVertexUvs[0][i * 2][2].y = e[2].y * 1 / 32
169 
170 		cube.geometry.faceVertexUvs[0][i * 2 + 1][0].x = e[1].x * 1 / 64
171 		cube.geometry.faceVertexUvs[0][i * 2 + 1][0].y = e[1].y * 1 / 32
172 		cube.geometry.faceVertexUvs[0][i * 2 + 1][1].x = e[3].x * 1 / 64
173 		cube.geometry.faceVertexUvs[0][i * 2 + 1][1].y = e[3].y * 1 / 32
174 		cube.geometry.faceVertexUvs[0][i * 2 + 1][2].x = e[2].x * 1 / 64
175 		cube.geometry.faceVertexUvs[0][i * 2 + 1][2].y = e[2].y * 1 / 32
176 	})

179 class Character extends THREE.Object3D {
180 	constructor({name, texture, fatness, tallness}) {
181 		super()
182 		// if (!fatness)
183 			fatness = 1
184 		// if (!tallness)
185 			tallness = 1
186 
187 		let _texture = new THREE.TextureLoader().load( texture.replace(" ", "_") )
188 		_texture.magFilter = THREE.NearestFilter;
189 		_texture.minFilter = THREE.NearestFilter;
190 
191 		let tMaterial = new THREE.MeshPhongMaterial({
192 			map: _texture,
193 			transparent: true,
194 			// opacity: 0.5,
195 			// color: 0xFF0000
196 		})
197 
198 		this.name = name
199 
200 		Object.keys(pieces).forEach( k => {
201 			let v = pieces[k]
202 
203 			let geometry = new THREE.BoxGeometry( v.size.width, v.size.height, v.size.depth )
204 			if (k == 'body')
205 				geometry = new THREE.BoxGeometry( v.size.width * fatness, v.size.height * tallness, v.size.depth * fatness)
206 
207 			let material = new THREE.MeshPhongMaterial( {
208 				color: 0xf50000,
209 				transparent: true,
210 				// wireframe: true
211 			} )
212 
213 			let cube = new THREE.Mesh( geometry, material )
214 			if (v.map) {
215 				cube = new THREE.Mesh( geometry, tMaterial )
216 				applyUV(v.map, cube)
217 			}
218 			cube.castShadow = true
219 			cube.receiveShadow = true
220 
221 			cube.userData.parent = this
222 
223 			this[k] = cube
224 			this[k]._size = v.size
225 			this[k]._position = v.position(fatness, tallness)
226 		})
227 
228 		Object.keys(pieces).forEach( k => {
229 			let v = pieces[k]
230 			this[k].position.set(
231 				this[k]._position(this.body).x,
232 				this[k]._position(this.body).y,
233 				this[k]._position(this.body).z
234 			)
235 			this.add(this[k])
236 		})
237 
238 		this.position.set(0, 1.1, 0)
239 		this.castShadow = true
240 		this.receiveShadow = true
241 
242 		this.tweens = []
243 
244 		this.left_leg.geometry.translate(0, -0.2, 0)
245 		this.left_leg.translateY(0.2)
246 		this.right_leg.geometry.translate(0, -0.2, 0)
247 		this.right_leg.translateY(0.2)
248 		this.left_arm.geometry.translate(0, -0.2, 0)
249 		this.left_arm.translateY(0.2)
250 		this.right_arm.geometry.translate(0, -0.2, 0)
251 		this.right_arm.translateY(0.2)
252 	}
253 
254 	update({clock, camera}) {
255 		let delta = clock.getDelta()
256 
257 		// this.head.lookAt( new THREE.Vector3(camera.position.x, camera.position.y - 1, camera.position.z) )
258 		this.tweens.forEach( t => t(delta) )
259 	}
260 
261 	walk() {
262 		let angle = Math.PI / 180
263 
264 		let f = (dt) => {
265 			if (this.left_leg.rotation._x > Math.PI / 8 || this.left_leg.rotation._x < -Math.PI / 8) {
266 				angle = -angle
267 			}
268 
269 			this.left_leg.rotateX(angle)
270 			this.right_leg.rotateX(-angle)
271 			this.left_arm.rotateX(-angle)
272 			this.right_arm.rotateX(angle)
273 
274 			let direction = new THREE.Vector3(0, 0, 1)
275 			this.translateOnAxis(direction, 0.01)
276 		}
277 
278 		if (typeof this.walking == "number") {
279 			this.tweens.splice(this.walking, 1)
280 			this.walking = undefined
281 
282 			let e = new THREE.Euler(0, 0, 0, "XYZ")
283 			this.left_leg.setRotationFromEuler(e)
284 			this.right_leg.setRotationFromEuler(e)
285 			this.left_arm.setRotationFromEuler(e)
286 			this.right_arm.setRotationFromEuler(e)
287 		}
288 		else {
289 			this.walking = this.tweens.push(f) - 1
290 		}
291 	}
292 }
